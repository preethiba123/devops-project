apiVersion: apps/v1
kind: Deployment
metadata:
  name: devops-app-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: devops-app
  template:
    metadata:
      labels:
        app: devops-app
    spec:
      containers:
        - name: devops-app-container
          image: devops-app:latest   # your local Docker image name
          imagePullPolicy: Never     # use local image from Docker Desktop
          ports:
            - containerPort: 3000

---
apiVersion: v1
kind: Service
metadata:
  name: devops-app-service
spec:
  selector:
    app: devops-app
  ports:
    - protocol: TCP
      port: 3000
      targetPort: 3000
  type: NodePort
